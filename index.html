let countries = ["Australia", "Germany", "US", "Canada"]; function getSmallest(arr){ let ansIdx = 0; for(let i = 0; i < arr.length; i++){ let ansLen = arr[ansIdx].length; let currLen = arr[i].length; if(currLen < ansLen){ ansIdx = i; } } return arr[ansIdx]; } console.log(getSmallest(countries)); // "US"
